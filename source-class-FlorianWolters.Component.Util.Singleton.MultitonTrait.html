18: trait MultitonTrait

20:     /**
21:      * Returns the *Multiton* instance of the class using this trait.
22:      *
23:      * @staticvar object[] $instances The *Multiton* instances of the classes
24:      *                                using this trait.
25:      *
26:      * @return object The *Multiton* instance.
27:      */
28:     public static function getInstance()
29:     {
30:         // This attribute has to be an array because it is shared across all
31:         // classes using this trait. Without an array, the first time an
32:         // instance is created and stored in this variable, all other calls
33:         // to the method getInstance() would return the same instance, no matter
34:         // what class is invoked.
35:         static $instances = [];
36: 
37:         // Get the name of the calling class. The calling class is the concrete
38:         // class using this trait.
39:         $className = \get_called_class();
40: 
41:         $arguments = \func_get_args();
42:         $key = $className . \serialize($arguments);
43: 
44:         if (false === isset($instances[$key])) {
45:             $instances[$key] = ReflectionUtils::createNewInstanceWithoutConstructor(
46:                 $className,
47:                 $arguments
48:             );
49:         }
50: 
51:         return $instances[$key];
52:     }
