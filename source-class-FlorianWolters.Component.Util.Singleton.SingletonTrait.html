19:     /**
20:      * Returns the *Singleton* instance of the class using this trait.
21:      *
22:      * @staticvar object[] $instances The *Singleton* instances of the classes
23:      *                                using this trait.
24:      *
25:      * @return object The *Singleton* instance.
26:      */
27:     final public static function getInstance()
28:     {
29:         // This attribute has to be an array because it is shared across all
30:         // classes using this trait. Without an array, the first time an
31:         // instance is created and stored in this variable, all other calls
32:         // to the method getInstance() would return the same instance, no matter
33:         // what class is invoked.
34:         static $instances = [];
35: 
36:         // Get the name of the calling class. The calling class is the concrete
37:         // class using this trait.
38:         $className = \get_called_class();
39: 
40:         if (false === isset($instances[$className])) {
41:             $arguments = \func_get_args();
42:             $instances[$className] = ReflectionUtils::createNewInstanceWithoutConstructor(
43:                 $className,
44:                 $arguments
45:             );
46:         }
47: 
48:         return $instances[$className];
49:     }
